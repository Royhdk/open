int a[20], b[20], c[20], A[3], B[3], C[3], d[20] = { 0 };
float n;

void random()
{
	srand((int)time(0));  // 产生随机种子
	d[0] = 1;
	d[1] = 2;
	d[2] = 3;
	for (int i = 3; i < 16; i++)
	{
		d[i] = rand() % 7 + 1;
	}
}
void show()
{
	cout << " |---+---+---+---+---+---+---+---+---+---+---+---+---+---+---+---|" << endl;
	cout << " |";
	for (int i = 0; i < 16; i++)
	{
		cout << " " << d[i] << " |";
	}
	cout << endl;
	cout << " |---+---+---+---+---+---+---+---+---+---+---+---+---+---+---+---|" << endl;
	cout << " |";
	for (int i = 0; i < 16; i++)
	{
		if (a[i] == 0) { cout << "   |"; }
		else cout << " " << a[i] << " |";
	}
	cout << endl;
	cout << " |";
	for (int i = 0; i < 16; i++)
	{
		if (b[i] == 0) { cout << "   |"; }
		else cout << " " << b[i] << " |";
	}
	cout << endl;
	cout << " |";
	for (int i = 0; i < 16; i++)
	{
		if (c[i] == 0) { cout << "   |"; }

	}
	cout << endl;
	cout << " |---+---+---+---+---+---+---+---+---+---+---+---+---+---+---+---|" << endl;
	cout << "缺页次数为：" << n << endl;
	cout << "缺页率：" << n / 16 << endl;
	system("pause");
}
void FIFO()
{
	a[0] = a[1] = a[2] = d[0];
	b[1] = b[2] = d[1];
	c[2] = d[2];
	A[0] = A[1] = A[2] = d[0];
	B[1] = B[2] = d[1];
	C[2] = d[2];
	n = 3;
	for (int i = 3; i < 16; i++)
	{
		if (d[i] == A[2] || d[i] == B[2] || d[i] == C[2])
		{
			a[i] = b[i] = c[i] = 0;
		}
		else
		{
			n++;
			if (A[0] == A[1] && A[1] == A[2])
			{
				a[i] = d[i];
				b[i] = B[2];
				c[i] = C[2];
				A[2] = d[i]; B[0] = B[1]; C[1] = C[2];
			}
			else if (B[0] == B[1] && B[1] == B[2])
			{
				a[i] = A[2];
				b[i] = d[i];
				c[i] = C[2];
				A[1] = A[2]; B[2] = d[i]; C[0] = C[1];
			}
			else if (C[0] == C[1] && C[1] == C[2])
			{
				a[i] = A[2];
				b[i] = B[2];
				c[i] = d[i];
				A[0] = A[1]; B[1] = B[2]; C[2] = d[i];
			}
		}
	}
	system("cls");
	cout << "LRU置换算法" << endl;
	show();

}
void LRU()
{

	a[0] = a[1] = a[2] = d[0];
	b[1] = b[2] = d[1];
	c[2] = d[2];
	A[0] = d[0];
	A[1] = d[1];
	A[2] = d[2];
	n = 3;
	for (int i = 3; i < 16; i++)
	{
		if (d[i] == A[0])
		{
			A[0] = A[1]; A[1] = A[2]; A[2] = d[i];
		}
		else if (d[i] == A[1])
		{
			A[1] = A[2]; A[2] = d[i];
		}
		else if (d[i] == A[2])
		{
			A[2] = d[i];
		}
		else
		{
			if (A[0] == a[i - 1]) { a[i] = d[i]; b[i] = b[i - 1]; c[i] = c[i - 1]; }
			else if (A[0] == b[i - 1] ){ a[i] = a[i-1]; b[i] = d[i]; c[i] = c[i - 1]; }
			else if (A[0] == c[i - 1]) { a[i] = a[i-1]; b[i] = b[i - 1]; c[i] = d[i]; }
			A[0] = A[1]; A[1] = A[2]; A[2] = d[i-1];
		}

	}
	system("cls");
	cout << "LRU置换算法" << endl;
	show();
}
